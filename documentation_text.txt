Password - NVARCHAR(255) NOT NULL (Hashed)

PhoneNumber - NVARCHAR(20)

Role - NVARCHAR(20) → 'Admin' / 'Landlord' / 'Tenant'
IsApproved - BIT DEFAULT 0 (Tenant=1 يئاقلت  ، Landlord رظتني Admin)

ProfileImage - NVARCHAR(500)

IsActive - BIT DEFAULT 1

CreatedAt - DATETIME DEFAULT GETDATE()

3.2 لودج Properties

Id - INT PRIMARY KEY IDENTITY

Title - NVARCHAR(150) NOT NULL

Description - NVARCHAR(MAX)

Price - DECIMAL(10,2) NOT NULL

Location - NVARCHAR(200) NOT NULL

PropertyType - NVARCHAR(50) → 'Apartment'/'Villa'/'Studio'/'Office'

RentalStatus - NVARCHAR(20) DEFAULT 'Available' → 'Available'/'Rented'/'Pending'
IsApproved - BIT DEFAULT 0 (Admin قفاوي)

IsActive - BIT DEFAULT 1

LandlordId - INT FK → Users(Id)

CreatedAt - DATETIME DEFAULT GETDATE()

UpdatedAt - DATETIME
